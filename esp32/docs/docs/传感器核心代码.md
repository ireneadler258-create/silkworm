## 一、CO2传感器 (JW01)

### 1.1 硬件连接

| ESP32引脚 | JW01引脚 | 说明 |
|----------|---------|------|
| GPIO15 | TX | CO2数据发送 |
| GPIO16 | RX | CO2数据接收 |
| VCC | 5V | 供电 |
| GND | GND | 地 |

### 1.2 核心代码

```python
# CO2传感器读取模块 co2_reader.py
from machine import UART
import time

# 缓存机制：避免频繁读取
_cached_co2 = None
_last_read_time = 0
_read_interval = 10  # 10秒读取间隔

def read_co2(guardian_obj=None, mqtt_client=None, force=False):
    """
    读取CO2浓度值（带缓存机制）
    
    Args:
        guardian_obj: 看门狗对象（保持读取期间喂狗）
        mqtt_client: MQTT客户端（保持连接）
        force: 强制读取标志
    
    Returns:
        CO2浓度值(ppm)或None
    """
    global _cached_co2, _last_read_time
    
    current_time = time.time()
    
    # 缓存检查：未过期直接返回
    if not force and (current_time - _last_read_time) < _read_interval:
        return _cached_co2
    
    # 初始化UART
    uart = UART(1, baudrate=9600, tx=15, rx=16, timeout=10)
    start_time = time.ticks_ms()
    
    # 读取循环（超时1.5秒）
    while time.ticks_diff(time.ticks_ms(), start_time) < 1500:
        # 喂狗保持系统活跃
        if guardian_obj: 
            guardian_obj.feed()
        
        # 检查MQTT消息
        if mqtt_client:
            try:
                mqtt_client.check_msg()
            except:
                pass
        
        # 检查数据可用
        if uart.any():
            raw = uart.read()
            # 查找帧头 0x2C
            index = raw.rfind(b'\x2c')
            if index != -1 and len(raw) >= index + 6:
                packet = raw[index : index + 6]
                # 校验和验证
                if (sum(packet[0:5]) & 0xFF) == packet[5]:
                    # 解析CO2值
                    ppm = (packet[1] << 8) | packet[2]
                    _cached_co2 = ppm
                    _last_read_time = current_time
                    uart.deinit()
                    return ppm
        
        time.sleep(0.1)
    
    uart.deinit()
    return _cached_co2
```

### 1.3 数据协议

```
帧格式: [0xFF] [CO2高8位] [CO2低8位] [校验和]

示例: FF 01 C2 01 C3
       ↓
CO2 = 0x01C2 = 450 ppm
```

## 二、GPS定位模块 (ATK-NEO6M)

### 2.1 硬件连接

| ESP32引脚 | NEO6M引脚 | 说明 |
|----------|----------|------|

| GPIO17 | RX | GPS数据接收 |
| VCC | 5V/3.3V | 供电 |
| GND | GND | 地 |

### 2.2 核心代码

```python
# GPS读取模块 gps_reader.py
from machine import UART
import time

# 缓存GPS数据
_cached_lon = None
_cached_lat = None
_cached_alt = None
_last_read_time = 0
_read_interval = 30  # 30秒读取间隔

def convert_to_degrees(raw_val, direction):
    """NMEA坐标转换为十进制度"""
    if not raw_val or not direction:
        return 0.0
    try:
        dot_idx = raw_val.find('.')
        if dot_idx < 2: return 0.0
        # ddmm.mmmm → dd + mm/60
        minutes_raw = float(raw_val[dot_idx - 2:])
        degrees_raw = float(raw_val[:dot_idx - 2])
        decimal_degrees = degrees_raw + (minutes_raw / 60)
        # 南纬/西经为负
        if direction in ['S', 'W']:
            decimal_degrees = -decimal_degrees
        return round(decimal_degrees, 6)
    except:
        return 0.0

def read_gps(guardian_obj=None, mqtt_client=None, force=False):
    """
    读取GPS数据（带缓存）
    
    Args:
        guardian_obj: 看门狗对象
        mqtt_client: MQTT客户端
        force: 强制读取
    
    Returns:
        (经度, 纬度, 海拔) 或缓存值
    """
    global _cached_lon, _cached_lat, _cached_alt, _last_read_time
    
    current_time = time.time()
    
    # 缓存检查
    if not force and (current_time - _last_read_time) < _read_interval:
        return _cached_lon, _cached_lat, _cached_alt
    
    print("[GPS] 开始读取...")
    
    # 初始化UART
    gps_uart = UART(1, baudrate=9600, tx=18, rx=17, timeout=10)
    buffer = ""
    start_time = time.ticks_ms()
    
    # 读取循环（超时2.2秒）
    while time.ticks_diff(time.ticks_ms(), start_time) < 2200:
        if guardian_obj: 
            guardian_obj.feed()
        if mqtt_client:
            try:
                mqtt_client.check_msg()
            except:
                pass
        
        # 读取数据
        if gps_uart.any():
            try:
                chunk = gps_uart.read().decode('ascii', 'ignore')
                buffer += chunk
                
                # 防止缓冲区溢出
                if len(buffer) > 2048:
                    buffer = buffer[-1024:]
                
                # 逐行解析
                if '\n' in buffer:
                    lines = buffer.split('\n')
                    buffer = lines.pop()
                    
                    for line in reversed(lines):
                        parts = line.split(',')
                        
                        # $GPGGA帧：定位数据
                        if '$GPGGA' in line and len(parts) > 9:
                            if parts[2] and parts[4]:
                                lat = convert_to_degrees(parts[2], parts[3])
                                lon = convert_to_degrees(parts[4], parts[5])
                                alt = float(parts[9]) if parts[9] else 0.0
                                _cached_lon, _cached_lat, _cached_alt = lon, lat, alt
                                _last_read_time = current_time
                                gps_uart.deinit()
                                return lon, lat, alt
                        
                        # $GPRMC帧：推荐定位数据
                        elif '$GPRMC' in line and len(parts) > 7:
                            if parts[2] == 'A' and parts[3] and parts[5]:
                                lat = convert_to_degrees(parts[3], parts[4])
                                lon = convert_to_degrees(parts[5], parts[6])
                                _cached_lon, _cached_lat, _cached_alt = lon, lat, 0.0
                                _last_read_time = current_time
                                gps_uart.deinit()
                                return lon, lat, 0.0
            except:
                pass
        
        time.sleep(0.05)
    
    gps_uart.deinit()
    return _cached_lon, _cached_lat, _cached_alt
```
